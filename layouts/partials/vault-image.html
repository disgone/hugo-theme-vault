{{- $src := .Get "src" -}} {{- $alt := .Get "alt" -}} {{- $title := .Get "title" -}} {{- $width := .Get "width" -}} {{- $height := .Get "height" -}} {{- $eager := .Get "eager" -}} {{- $format := .Get "format" -}} {{- $sizes := .Get "sizes" | default "(min-width: 60rem) 60rem, 100vw" -}} {{- $longdesc := .Get "longdesc" -}} {{- $describedby := .Get "describedby" -}} {{/* Auto-detect dimensions for local images if not provided */}} {{- $img := resources.Get $src -}} {{- $origWidth := $width -}} {{- $origHeight := $height -}} {{- if and $img (not $width) (not $height) -}} {{- $origWidth = $img.Width -}} {{- $origHeight = $img.Height -}} {{- end -}} {{/* Generate responsive image candidates for local images */}} {{- $srcWithFormat := $src -}} {{- $webpSrcset := slice -}} {{- $fallbackSrcset := slice -}} {{- $fallbackSrc := $src -}} {{- if $img -}} {{- $candidateWidths := slice -}} {{- range (slice 480 768 1024 1440 $img.Width) -}} {{- if and (le . $img.Width) (not (in $candidateWidths .)) -}} {{- $candidateWidths = $candidateWidths | append . -}} {{- end -}} {{- end -}} {{- range $candidateWidths -}} {{- $fallbackVariant := $img.Resize (printf "%dx" .) -}} {{- $webpVariant := $img.Resize (printf "%dx webp" .) -}} {{- $fallbackSrcset = $fallbackSrcset | append (printf "%s %dw" $fallbackVariant.RelPermalink .) -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $webpVariant.RelPermalink .) -}} {{- if eq . $img.Width -}} {{- $fallbackSrc = $fallbackVariant.RelPermalink -}} {{- end -}} {{- end -}} {{- end -}} {{- if $img -}} {{- if or $alt $title -}} {{- $fallbackContent := "" -}} {{- if $alt -}} {{- $fallbackContent = $alt -}} {{- else if $title -}} {{- $fallbackContent = "Image" -}} {{- end -}} {{- end -}} {{- else -}} {{- end -}}
